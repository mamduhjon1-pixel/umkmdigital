rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() { return request.auth != null; }
    function me() { return get(/databases/$(database)/documents/users/$(request.auth.uid)); }
    function myRole() { return signedIn() && me().exists() ? me().data.role : null; }
    function isAdminUtama() { return signedIn() && myRole() == 'admin'; }
    function isAdmin() { return signedIn() && myRole() in ['admin', 'sub_admin']; }
    function isSeller() { return signedIn() && myRole() == 'seller'; }
    function isBuyer() { return signedIn() && myRole() == 'buyer'; }
    function isOwner(uid) { return signedIn() && uid == request.auth.uid; }
    function onlyChanged(keys) { return request.resource.data.diff(resource.data).changedKeys().hasOnly(keys); }

    match /users/{uid} {
      allow get: if signedIn() && (isOwner(uid) || isAdmin());
      // Kode saat ini membaca collection users secara global untuk dashboard/chat.
      // Karena itu list hanya untuk user login. Untuk lebih ketat, refactor query per role dulu.
      allow list: if signedIn();

      // Registrasi publik hanya boleh membuat buyer/seller untuk dirinya sendiri.
      // Admin utama boleh membuat sub_admin dari dashboard.
      allow create: if (
          signedIn()
          && isOwner(uid)
          && request.resource.data.uid == uid
          && request.resource.data.role in ['buyer', 'seller']
          && !(request.resource.data.keys().hasAny(['permissions']))
        ) || (
          isAdminUtama()
          && request.resource.data.uid == uid
          && request.resource.data.role == 'sub_admin'
          && request.resource.data.status == 'active'
        );

      // User boleh update profil/alamat/token sendiri, tapi tidak boleh naik role/status/permission.
      allow update: if isAdmin()
        || (
          isOwner(uid)
          && !request.resource.data.diff(resource.data).changedKeys().hasAny([
            'role', 'status', 'permissions', 'commissionBlocked', 'isDeleted'
          ])
        );
      allow delete: if isAdmin();
    }

    match /products/{productId} {
      allow read: if true;
      allow create: if signedIn()
        && isSeller()
        && request.resource.data.sellerId == request.auth.uid;
      allow update: if isAdmin()
        || (signedIn() && resource.data.sellerId == request.auth.uid);
      allow delete: if isAdmin();
    }

    match /orders/{orderId} {
      // Kode saat ini memakai onSnapshot(collection('orders')) lalu filter di client.
      // Agar app tidak patah, semua user login bisa list order. Data sensitif sebaiknya difilter di query sebelum rules diperketat.
      allow read: if signedIn();
      allow create: if signedIn()
        && isBuyer()
        && request.resource.data.buyerId == request.auth.uid;
      allow update: if isAdmin()
        || (signedIn() && resource.data.buyerId == request.auth.uid)
        || (signedIn() && resource.data.sellerId == request.auth.uid);
      allow delete: if isAdmin();
    }

    match /reviews/{reviewId} {
      allow read: if true;
      allow create: if signedIn()
        && request.resource.data.buyerId == request.auth.uid;
      allow update, delete: if isAdmin();
    }

    match /withdrawals/{withdrawalId} {
      allow read: if signedIn();
      allow create: if signedIn()
        && isSeller()
        && request.resource.data.sellerId == request.auth.uid;
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }

    match /seller_wallets/{sellerId} {
      // Kode dashboard membaca seluruh seller_wallets; dibuka untuk user login agar tidak error.
      allow read: if signedIn();
      // Dibutuhkan kode saat ini saat register seller dan transaksi client-side.
      allow create: if isAdmin() || (signedIn() && sellerId == request.auth.uid);
      allow update: if isAdmin() || (signedIn() && sellerId == request.auth.uid);
      allow delete: if isAdmin();
    }

    match /wallet_transactions/{txId} {
      allow read: if signedIn();
      allow create: if signedIn();
      allow update, delete: if isAdmin();
    }

    match /komisi_tagihan/{billId} {
      allow read: if signedIn();
      allow create, update: if isAdmin()
        || (signedIn() && request.resource.data.sellerId == request.auth.uid)
        || (signedIn() && resource.data.sellerId == request.auth.uid);
      allow delete: if isAdmin();
    }

    match /admin_wallets/{walletId} {
      allow read: if signedIn();
      allow write: if isAdmin();
    }

    match /admin_commission_transactions/{txId} {
      allow read: if signedIn();
      allow create, update, delete: if isAdmin();
    }

    match /admin_settings/{settingId} {
      allow read: if signedIn();
      allow write: if isAdmin();
    }

    match /notifications/{notifId} {
      allow read: if signedIn() && (
        isAdmin()
        || resource.data.userId == request.auth.uid
        || resource.data.role == myRole()
      );
      allow create: if signedIn() && (
        request.resource.data.userId == request.auth.uid
        || request.resource.data.userId is string
        || request.resource.data.role in ['admin', 'seller', 'buyer', 'chat']
      );
      allow update: if signedIn() && (
        isAdmin()
        || resource.data.userId == request.auth.uid
      );
      allow delete: if isAdmin();
    }

    match /chats/{chatId} {
      function isParticipant() {
        return signedIn() && request.auth.uid in resource.data.participants;
      }
      function willBeParticipant() {
        return signedIn() && request.auth.uid in request.resource.data.participants;
      }
      allow read: if signedIn();
      allow create: if willBeParticipant() || isAdmin();
      allow update: if isParticipant() || willBeParticipant() || isAdmin();
      allow delete: if isParticipant() || isAdmin();

      match /messages/{messageId} {
        function parentChat() {
          return get(/databases/$(database)/documents/chats/$(chatId));
        }
        function parentParticipant() {
          return signedIn() && request.auth.uid in parentChat().data.participants;
        }
        allow read: if parentParticipant() || isAdmin();
        allow create: if parentParticipant()
          && request.resource.data.senderId == request.auth.uid;
        allow update: if parentParticipant() || isAdmin();
        allow delete: if parentParticipant() || isAdmin();
      }
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
